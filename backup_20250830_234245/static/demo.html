API Integration Example

  method: 'POST',
  headers: {
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    certificateId: 'NANO-CERT-2025-001'
  })
});

const result = await response.json();
console.log('Verification result:', result);
